    nyt_last_name varchar,
    nyt_date_confirmed varchar,
    nyt_date_nominated varchar,
    nyt_date_sworn_in varchar,
    nyt_first_term varchar,
    nyt_last_term varchar,
    nyt_first_naturalcourt varchar,
    nyt_last_naturalcourt varchar,
    nyt_family_status varchar,
    nyt_non_white bool,
    nyt_birth_year varchar,
    nyt_death_year varchar,
    nyt_religion varchar,
    nyt_seat varchar,
    nyt_preceded_by int,
    nyt_succeeded_by int
);
